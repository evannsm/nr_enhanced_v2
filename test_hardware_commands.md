# First non-pyjoules except for helix spin, then pyjoules except for helix spin.
# Separately after we'll do helix spin with and without pyjoules.

## Part 1: All Non-PyJoules Except Helix Spin

```bash
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory hover      --double-speed --hover-mode 1 --log
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory yaw_only   --double-speed --log
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory circle_horz --double-speed --log
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory circle_horz --double-speed --spin --log
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory circle_vert --double-speed --log
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory fig8_horz   --double-speed --log
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory fig8_vert   --double-speed --log
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory fig8_vert   --double-speed --short --log
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory helix       --double-speed --log
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory sawtooth    --double-speed --log
ros2 run nr_enhanced_v2 run_node --platform hw --trajectory triangle    --double-speed --log
```
